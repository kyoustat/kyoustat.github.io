For $X \sim \chi^2 (k)$, we have
\begin{itemize}
	\item \textbf{CLT}
	\begin{equation*}
	\frac{X - k}{\sqrt{2k}} \sim N(0,1)
	\end{equation*}
	\item \textbf{Fisher (1922)}
	\begin{equation*}
	\sqrt{2X} \sim N(\sqrt{2k-1}, 1)
	\end{equation*}
	\item \textbf{Wilson-Hilferty (1931)}
	\begin{equation*}
	\left({\frac{X}{k}}\right)^{1/3} \sim N(1-\frac{2}{9k}, \frac{2}{9k})
	\end{equation*}
\end{itemize}
